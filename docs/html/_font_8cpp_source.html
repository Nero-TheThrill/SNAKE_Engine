9static std::vector<char32_t> UTF8ToCodepoints(const std::string& text)
10{
11 std::vector<char32_t> result;
12 size_t i = 0;
13
14 while (i < text.size())
15 {
16 char32_t cp = 0;
17 unsigned char byte = text[i];
18
19 if ((byte & 0b10000000) == 0)
20 {
21 cp = byte;
22 ++i;
23 }
24 else if ((byte & 0b11100000) == 0b11000000)
25 {
26 cp = (byte & 0b00011111) << 6;
27 cp |= (text[i + 1] & 0b00111111);
28 i += 2;
29 }
30 else if ((byte & 0b11110000) == 0b11100000)
31 {
32 cp = (byte & 0b00001111) << 12;
33 cp |= (text[i + 1] & 0b00111111) << 6;
34 cp |= (text[i + 2] & 0b00111111);
35 i += 3;
36 }
37 else if ((byte & 0b11111000) == 0b11110000)
38 {
39 cp = (byte & 0b00000111) << 18;
40 cp |= (text[i + 1] & 0b00111111) << 12;
41 cp |= (text[i + 2] & 0b00111111) << 6;
42 cp |= (text[i + 3] & 0b00111111);
43 i += 4;
44 }
45 else
46 {
47 ++i;
48 continue;
49 }
50
51 result.push_back(cp);
52 }
53
54 return result;
55}

101bool Font::TryBakeGlyph(char32_t c)
102{
103 if (glyphs.find(c) != glyphs.end())
104 return true;
105
106 if (FT_Load_Char(face, c, FT_LOAD_RENDER))
107 {
108 SNAKE_ERR("FT_Load_Char failed for: U+" << std::hex << (int)c);
109 return false;
110 }
111
112 FT_GlyphSlot g = face->glyph;
113 int w = g->bitmap.width;
114 int h = g->bitmap.rows;
115 bool hasBitmap = (w > 0 && h > 0);
116 bool hasAdvance = (g->advance.x > 0);
117
118 if (!hasBitmap && !hasAdvance)
119 {
120 SNAKE_WRN("Glyph has no bitmap or advance: U+" << std::hex << (int)c);
121 return false;
122 }
123
124 const int padding = 1;
125 const int safeW = std::max(1, w);
126 const int safeH = std::max(1, h);
127 const int cellW = safeW + padding * 2;
128 const int cellH = safeH + padding * 2;
129
130 if (nextX + cellW > atlasTexture->GetWidth())
131 {
132 nextX = 0;
133 nextY += maxRowHeight;
134 maxRowHeight = 0;
135 }
136
137 if (nextY + cellH > atlasTexture->GetHeight())
138 {
139 ExpandAtlas();

141 }
142
143 int drawX = nextX + padding;
144 int drawY = nextY + padding;
145
146 if (hasBitmap && g->bitmap.buffer)
147 {
148 glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
149 glTextureSubImage2D(
150 atlasTexture->GetID(), 0,
151 drawX, drawY, w, h,
152 GL_RED, GL_UNSIGNED_BYTE,
153 g->bitmap.buffer
154 );
155 }
156 else
157 {
158 unsigned char dummy = 0;
159 glTextureSubImage2D(
160 atlasTexture->GetID(), 0,
161 drawX, drawY, 1, 1,
162 GL_RED, GL_UNSIGNED_BYTE,
163 &dummy
164 );
165 }
166
167 Glyph glyph;
168 glyph.size = { w, h };
169 glyph.bearing = { g->bitmap_left, g->bitmap_top };
170 glyph.advance = g->advance.x;
171
172 glyph.uvTopLeft = {
173 static_cast<float>(drawX) / atlasTexture->GetWidth(),
174 static_cast<float>(drawY) / atlasTexture->GetHeight()
175 };
176 glyph.uvBottomRight = {
177 static_cast<float>(drawX + safeW) / atlasTexture->GetWidth(),
178 static_cast<float>(drawY + safeH) / atlasTexture->GetHeight()
179 };
180
181 glyphs[c] = glyph;
182
183 nextX += cellW;
184 maxRowHeight = std::max(maxRowHeight, cellH);
185
186 return true;

206glm::vec2 Font::GetTextSize(const std::string& text) const
207{
208 float maxWidth = 0.0f;
209 float currentLineWidth = 0.0f;
210 int lineCount = 1;
211
212 std::vector<char32_t> codepoints = UTF8ToCodepoints(text);
213 for (char32_t c : codepoints)
214 {
215 if (c == U'\n')

217 maxWidth = std::max(maxWidth, currentLineWidth);
