20{

465void RenderManager::BuildRenderMap(const std::vector<Object*>& source, Camera2D* camera)
466{
467 for (auto* obj : source)
468 {
469 if (!obj || !obj->IsVisible())
470 continue;
471
472 Material* material = obj->GetMaterial();
473 Mesh* mesh = obj->GetMesh();
474 SpriteAnimator* spriteAnimator = obj->GetSpriteAnimator();
475
476 SpriteSheet* spritesheet = spriteAnimator ? spriteAnimator->GetSpriteSheet() : nullptr;
477 Shader* shader = material ? material->GetShader() : nullptr;
478
479 if (!material || !mesh || !shader)
480 continue;
481
482 uint8_t layer = renderLayerManager.GetLayerID(obj->GetRenderLayerTag()).value_or(0);
483 if (layer >= RenderLayerManager::MAX_LAYERS)
484 {
485 SNAKE_WRN("render skipped - invalid layer\n");
486 continue;
487 }
488
489 InstanceBatchKey key{ mesh, material, spritesheet };
490 renderMap[layer][shader][key].emplace_back(obj, camera);
491 }
492}
