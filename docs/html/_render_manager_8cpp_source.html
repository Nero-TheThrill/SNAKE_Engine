37void FrustumCuller::CullVisible(const Camera2D& camera, const std::vector<Object*>& allObjects,
38 std::vector<Object*>& outVisibleList, glm::vec2 viewportSize)
39{
40 outVisibleList.clear();
41 for (Object* obj : allObjects)
42 {
43 if (!obj->IsAlive() || !obj->IsVisible())
44 continue;
45 if (obj->ShouldIgnoreCamera())
46 {
47 outVisibleList.push_back(obj);

50 const glm::vec2& pos = obj->GetTransform2D().GetPosition();

251void RenderManager::BuildRenderMap(const std::vector<Object*>& source, Camera2D* camera)
252{
253 for (auto* obj : source)
254 {
255 if (!obj || !obj->IsVisible())
256 continue;
257
258 Material* material = obj->GetMaterial();
259 Mesh* mesh = obj->GetMesh();
260 Shader* shader = material ? material->GetShader() : nullptr;
261
262 if (!material || !mesh || !shader)
263 continue;
264
265 uint8_t layer = obj->GetRenderLayer();
266 if (layer >= RenderLayerManager::MAX_LAYERS)
267 {
268 SNAKE_WRN("render skipped - invalid layer\n");
269 continue;
270 }
271
272 InstanceBatchKey key{ mesh, material };
273 renderMap[layer][shader][key].emplace_back(obj, camera);
274 }

289 {

293 std::vector<glm::mat4> transforms;

298 colors.reserve(batch.size());

301 for (const auto& [obj, camera] : batch)
302 {
303 glm::mat4 model = obj->GetTransform2DMatrix();
304 glm::vec2 flip = obj->GetUVFlipVector();

311 uvOffsets.push_back(obj->GetAnimator()->GetUVOffset());
312 uvScales.push_back(obj->GetAnimator()->GetUVScale());

319 }
320

323
324 if (material != lastMaterial)

326 material->Bind();

328 }

345 material->SetTexture("u_Texture", batch.front().first->GetAnimator()->GetTexture());

348

350 material->SendUniforms();
351
352 key.mesh->BindVAO();
353 material->UpdateInstanceBuffer(transforms, colors, uvOffsets, uvScales);
354 key.mesh->DrawInstanced(static_cast<GLsizei>(transforms.size()));
355 material->UnBind();
356 });
357 }
358 else

361 {
362
