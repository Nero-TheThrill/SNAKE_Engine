65RenderMap RenderManager::BuildRenderMap(const std::vector<GameObject*>& source)
66{
67 RenderMap renderMap;
68
69 for (auto* obj : source)
70 {
71 if (!obj || !obj->IsVisible())
72 continue;
73
74 Material* material = obj->GetMaterial();
75 Mesh* mesh = obj->GetMesh();
76 Shader* shader = material ? material->GetShader() : nullptr;
77
78 if (!material || !mesh || !shader)
79 continue;
80
81 uint8_t layer = obj->GetRenderLayer();
82 if (layer >= RenderLayerManager::MAX_LAYERS)
83 {
84 SNAKE_WRN("render skipped - invalid layer\n");
85 continue;
86 }
87
88 InstanceBatchKey key{ mesh, material };
89 renderMap[layer][shader][key].push_back(obj);
90 }
91
92 return renderMap;
93}

96{
97 Material* lastMaterial = nullptr;
98 Shader* lastShader = nullptr;
99
100 for (uint8_t layer = 0; layer < renderMap.size(); ++layer)
101 {

103
104 for (const auto& [shader, batchMap] : _shaderMap)
105 {

107 {
108 if (batch.front()->CanBeInstanced())
109 {
110 Submit([=]() mutable {
111 std::vector<glm::mat4> transforms;
112 transforms.reserve(batch.size());
113 for (auto* obj : batch)
114 transforms.push_back(obj->GetTransform2DMatrix());
115
116 Material* material = key.material;
117 Shader* currentShader = material->GetShader();
118
119 if (material != lastMaterial)
120 {
121 material->Bind();
122 lastMaterial = material;
123 }
124

128 lastShader = currentShader;

134 key.mesh->BindVAO();

136 key.mesh->DrawInstanced(static_cast<GLsizei>(transforms.size()));
137 material->UnBind();
138 });
139 }

180{

273 materialMap[tag] = std::move(material);
